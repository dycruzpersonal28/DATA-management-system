import { createClient } from '@supabase/supabase-js'

// Service-role client â€” server only, never exposed to the browser
const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!,
  { auth: { autoRefreshToken: false, persistSession: false } }
)

export async function createAppUser({
  shopId, name, email, password, role,
}: {
  shopId: string
  name: string
  email: string
  password: string
  role: string
}) {
  console.log('[createAppUser] called with:', { shopId, name, email, role })
  console.log('[createAppUser] service key loaded:', !!process.env.SUPABASE_SERVICE_ROLE_KEY)

  const { data: authData, error: authError } = await supabaseAdmin.auth.admin.createUser({
    email,
    password,
    email_confirm: true,
    user_metadata: { name, role, shop_id: shopId },
  })

  console.log('[createAppUser] auth result:', authData?.user?.id, 'error:', authError?.message)

  if (authError) return { error: authError.message }

  const { data, error: dbError } = await supabaseAdmin
    .from('app_users')
    .insert({
      shop_id: shopId,
      auth_user_id: authData.user.id,
      name,
      email,
      role,
      is_active: true,
    })
    .select()
    .single()

  console.log('[createAppUser] db result:', data, 'error:', dbError?.message)

  if (dbError) {
    await supabaseAdmin.auth.admin.deleteUser(authData.user.id)
    return { error: dbError.message }
  }

  return { data }
}

// â”€â”€ Update name / role, and optionally reset password â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
export async function updateAppUser({
  appUserId,
  authUserId,
  name,
  role,
  password,
}: {
  appUserId: string
  authUserId: string | null
  name: string
  role: string
  password?: string
}) {
  const { error: dbError } = await supabaseAdmin
    .from('app_users')
    .update({ name, role })
    .eq('id', appUserId)

  if (dbError) return { error: dbError.message }

  if (password && authUserId) {
    const { error: pwError } = await supabaseAdmin.auth.admin.updateUserById(authUserId, {
      password,
      user_metadata: { name, role },
    })
    if (pwError) return { error: pwError.message }
  }

  return { data: { id: appUserId, name, role } }
}

export async function deleteAppUser({
  appUserId,
  authUserId,
}: {
  appUserId: string
  authUserId: string | null
}) {
  const { error: dbError } = await supabaseAdmin
    .from('app_users')
    .delete()
    .eq('id', appUserId)

  if (dbError) return { error: dbError.message }

  if (authUserId) {
    await supabaseAdmin.auth.admin.deleteUser(authUserId)
  }

  return { data: { deleted: appUserId } }
}

function TerminalsSection({ shopId }: { shopId: string }) {
  const supabase = createClient()
  const [terminals, setTerminals] = useState<any[]>([])
  const [adding, setAdding] = useState(false)
  const [editingId, setEditingId] = useState<string | null>(null)
  const [form, setForm] = useState({ name: '' })
  const [loading, setLoading] = useState(false)

  useEffect(() => {
    if (!shopId) return
    supabase.from('pos_terminals').select('*').eq('shop_id', shopId).order('created_at')
      .then(({ data }) => setTerminals(data || []))
  }, [shopId])

  async function handleSave() {
    if (!form.name.trim()) return toast.error('Name required')
    setLoading(true)
    if (editingId) {
      await supabase.from('pos_terminals').update({ name: form.name }).eq('id', editingId)
      setTerminals(prev => prev.map(t => t.id === editingId ? { ...t, name: form.name } : t))
      toast.success('Updated')
      setEditingId(null)
    } else {
      const { data, error } = await supabase.from('pos_terminals')
        .insert({ shop_id: shopId, name: form.name, is_active: true })
        .select().single()
      if (error) { toast.error('Failed'); setLoading(false); return }
      setTerminals(prev => [...prev, data])
      toast.success('Terminal added')
      setAdding(false)
    }
    setForm({ name: '' })
    setLoading(false)
  }

  async function handleDelete(id: string) {
    await supabase.from('pos_terminals').delete().eq('id', id)
    setTerminals(prev => prev.filter(t => t.id !== id))
    toast.success('Deleted')
  }

  async function handleToggle(t: any) {
    const val = !t.is_active
    await supabase.from('pos_terminals').update({ is_active: val }).eq('id', t.id)
    setTerminals(prev => prev.map(p => p.id === t.id ? { ...p, is_active: val } : p))
  }

  return (
    <div>
      <div className="flex items-center justify-between mb-3">
        <div>
          <h3 className="text-base font-semibold text-gray-900">POS Terminals</h3>
          <p className="text-xs text-gray-500 mt-0.5">Create named terminals so you can track which POS processed each transaction.</p>
        </div>
        {!adding && !editingId && (
          <Button variant="outline" size="sm" onClick={() => setAdding(true)}>
            <Plus className="w-3.5 h-3.5 mr-1.5" /> Add Terminal
          </Button>
        )}
      </div>

      {(adding || editingId) && (
        <div className="bg-gray-50 border border-gray-200 rounded-xl p-4 mb-3 flex gap-2">
          <Input
            value={form.name}
            onChange={e => setForm({ name: e.target.value })}
            placeholder="e.g. Counter 1, Drive-Thru, Bar"
            onKeyDown={e => e.key === 'Enter' && handleSave()}
            autoFocus
          />
          <Button size="sm" onClick={handleSave} disabled={loading}><Check className="w-3.5 h-3.5" /></Button>
          <Button size="sm" variant="outline" onClick={() => { setAdding(false); setEditingId(null); setForm({ name: '' }) }}><X className="w-3.5 h-3.5" /></Button>
        </div>
      )}

      <div className="space-y-2">
        {terminals.length === 0 && !adding && (
          <div className="text-center py-6 text-gray-400 text-xs">No terminals yet.</div>
        )}
        {terminals.map(t => (

            <div className={`w-8 h-8 rounded-lg flex items-center justify-center flex-shrink-0 ${t.is_active ? 'bg-indigo-100' : 'bg-gray-100'}`}>
              <Monitor className={`w-4 h-4 ${t.is_active ? 'text-indigo-600' : 'text-gray-400'}`} />
            </div>
            <div className="flex-1 min-w-0">

            </div>

const EMPTY_FORM: UserForm = { name: '', email: '', password: '', newPassword: '', role: 'cashier' }

function UserManagementSection({ shopId }: { shopId: string }) {

  const [users, setUsers] = useState<AppUser[]>([])

      if (result.error) { toast.error(result.error); setLoading(false); return }
      setUsers(prev => [...prev, result.data as AppUser])
      toast.success(`${form.name} created â€” they can log in immediately`)
    }

    setLoading(false)
    closeModal()
  }

  async function handleToggleActive(user: AppUser) {
    const val = !user.is_active
    await supabase.from('app_users').update({ is_active: val }).eq('id', user.id)
    setUsers(prev => prev.map(u => u.id === user.id ? { ...u, is_active: val } : u))
    toast.success(val ? 'User activated' : 'User deactivated')
  }

  async function handleDelete(user: AppUser) {
    if (!confirm(`Remove ${user.name}? This will also delete their login.`)) return
    const result = await deleteAppUser({ appUserId: user.id, authUserId: user.auth_user_id })
    if (result.error) { toast.error(result.error); return }
    setUsers(prev => prev.filter(u => u.id !== user.id))
    toast.success('User deleted')
  }

  async function handleResetPassword() {
    if (!resetUser) return
    if (resetPassword.length < 6) return toast.error('Password must be at least 6 characters')
    setResetLoading(true)
    const result = await updateAppUser({
      appUserId: resetUser.id,
      authUserId: resetUser.auth_user_id,
      name: resetUser.name,
      role: resetUser.role,
      password: resetPassword,
    })
    if (result.error) { toast.error(result.error); setResetLoading(false); return }
    toast.success(`Password reset for ${resetUser.name}`)
    setResetUser(null)
    setResetPassword('')
    setResetLoading(false)
  }

  return (
    <div>
      {/* Header */}
      <div className="flex items-center justify-between mb-3">
        <div>
          <h3 className="text-base font-semibold text-gray-900">User Management</h3>
          <p className="text-xs text-gray-500 mt-0.5">
            Create users with email and password. No email verification â€” they can log in right away.
          </p>
        </div>
        <Button size="sm" onClick={openCreate} className="bg-indigo-600 hover:bg-indigo-700 text-white">
          <UserPlus className="w-3.5 h-3.5 mr-1.5" /> Add User
        </Button>
      </div>

      {/* User list */}
      <div className="space-y-2">
        {users.length === 0 && (
          <div className="text-center py-10 text-gray-400 text-xs">
            No users yet. Click <span className="font-medium text-gray-500">Add User</span> to create one.
          </div>
        )}
        {users.map(user => (
          <div key={user.id} className={`bg-white rounded-xl border px-4 py-3 flex items-center gap-3 transition-opacity ${user.is_active ? 'border-gray-200' : 'border-gray-100 opacity-60'}`}>
            <div className={`w-9 h-9 rounded-xl flex items-center justify-center flex-shrink-0 font-semibold text-sm ${user.is_active ? 'bg-indigo-100 text-indigo-700' : 'bg-gray-100 text-gray-400'}`}>
              {user.name.charAt(0).toUpperCase()}
            </div>
            <div className="flex-1 min-w-0">
              <div className="flex items-center gap-2">
                <p className={`text-sm font-semibold truncate ${user.is_active ? 'text-gray-900' : 'text-gray-400'}`}>{user.name}</p>
                <span className={`px-2 py-0.5 rounded-full text-xs font-medium flex-shrink-0 ${roleColor(user.role)}`}>{user.role}</span>
              </div>
              <div className="flex items-center gap-1 mt-0.5">
                <Mail className="w-3 h-3 text-gray-300 flex-shrink-0" />
                <p className="text-xs text-gray-400 truncate">{user.email}</p>
              </div>
            </div>
            <div className="flex items-center gap-2 flex-shrink-0">
              <button
                onClick={() => handleToggleActive(user)}
                className={`px-3 py-1 rounded-lg text-xs font-medium transition-colors ${user.is_active ? 'bg-green-50 text-green-600 hover:bg-green-100' : 'bg-gray-100 text-gray-400 hover:bg-gray-200'}`}
              >
                {user.is_active ? 'Active' : 'Inactive'}
              </button>
              {/* Reset password */}
              <button
                onClick={() => { setResetUser(user); setResetPassword(''); setShowResetPassword(false) }}
                className="p-1.5 rounded-lg hover:bg-blue-50 text-gray-400 hover:text-blue-500"
                title="Reset password"
              >
                <KeyRound className="w-3.5 h-3.5" />
              </button>
              <button onClick={() => openEdit(user)} className="p-1.5 rounded-lg hover:bg-gray-100 text-gray-400">
                <Pencil className="w-3.5 h-3.5" />
              </button>
              <button onClick={() => handleDelete(user)} className="p-1.5 rounded-lg hover:bg-red-50 text-gray-400 hover:text-red-500">
                <Trash2 className="w-3.5 h-3.5" />
              </button>
            </div>
          </div>
        ))}
      </div>

      {/* Create / Edit Modal */}
      {showModal && (
        <div className="fixed inset-0 z-50 flex items-center justify-center">
          <div className="absolute inset-0 bg-black/30 backdrop-blur-sm" onClick={closeModal} />
          <div className="relative bg-white rounded-2xl shadow-xl w-full max-w-md mx-4 p-6">
            <div className="flex items-center justify-between mb-5">
              <div>
                <h4 className="text-base font-semibold text-gray-900">
                  {editingUser ? 'Edit User' : 'Create User'}
                </h4>
                <p className="text-xs text-gray-500 mt-0.5">
                  {editingUser ? 'Update name or role.' : 'Set a temporary password â€” user can change it after logging in.'}
                </p>
              </div>
              <button onClick={closeModal} className="p-1.5 rounded-lg hover:bg-gray-100 text-gray-400">
                <X className="w-4 h-4" />
              </button>
            </div>

            <div className="space-y-4">
              {/* Name */}
              <div>
                <label className="block text-xs font-medium text-gray-700 mb-1.5">Full Name</label>
                <Input
                  value={form.name}
                  onChange={e => setForm(f => ({ ...f, name: e.target.value }))}
                  placeholder="e.g. Sarah Johnson"
                  autoFocus
                />
              </div>

              {/* Email â€” create only */}
              <div>
                <label className="block text-xs font-medium text-gray-700 mb-1.5">Email Address</label>
                <div className="relative">
                  <Mail className="absolute left-3 top-1/2 -translate-y-1/2 w-3.5 h-3.5 text-gray-400" />
                  <Input
                    type="email"
                    value={form.email}
                    onChange={e => setForm(f => ({ ...f, email: e.target.value }))}
                    placeholder="user@example.com"
                    className="pl-9"
                    disabled={!!editingUser}
                  />
                </div>
                {editingUser && <p className="text-xs text-gray-400 mt-1">Email cannot be changed after creation.</p>}
              </div>

              {/* Temporary password â€” create only */}
              {!editingUser && (
                <div>
                  <label className="block text-xs font-medium text-gray-700 mb-1.5">
                    Temporary Password
                    <span className="ml-1.5 text-gray-400 font-normal">â€” user can change after login</span>
                  </label>
                  <div className="relative">
                    <KeyRound className="absolute left-3 top-1/2 -translate-y-1/2 w-3.5 h-3.5 text-gray-400" />
                    <Input
                      type={showPassword ? 'text' : 'password'}
                      value={form.password}
                      onChange={e => setForm(f => ({ ...f, password: e.target.value }))}
                      placeholder="Min. 6 characters"
                      className="pl-9 pr-9"
                    />
                    <button
                      type="button"
                      onClick={() => setShowPassword(p => !p)}
                      className="absolute right-2.5 top-1/2 -translate-y-1/2 text-gray-400 hover:text-gray-600"
                    >
                      {showPassword ? <EyeOff className="w-3.5 h-3.5" /> : <Eye className="w-3.5 h-3.5" />}
                    </button>
                  </div>
                </div>
              )}

              {/* Role */}
              <div>
                <label className="block text-xs font-medium text-gray-700 mb-1.5">Role</label>
                <div className="relative">
                  <button
                    type="button"
                    onClick={() => setShowRoleDropdown(p => !p)}
                    className="w-full flex items-center justify-between px-3 py-2 border border-gray-200 rounded-lg text-sm text-gray-700 bg-white hover:bg-gray-50 transition-colors"
                  >
                    <span className={`px-2 py-0.5 rounded-full text-xs font-medium ${roleColor(form.role)}`}>{form.role}</span>
                    <ChevronDown className="w-3.5 h-3.5 text-gray-400" />
                  </button>
                  {showRoleDropdown && (
                    <div className="absolute z-10 top-full mt-1 left-0 right-0 bg-white border border-gray-200 rounded-xl shadow-lg overflow-hidden">
                      {ROLES.map(r => (
                        <button key={r} type="button"
                          onClick={() => { setForm(f => ({ ...f, role: r })); setShowRoleDropdown(false) }}
                          className={`w-full flex items-center gap-3 px-4 py-2.5 text-sm hover:bg-gray-50 transition-colors ${form.role === r ? 'bg-gray-50' : ''}`}
                        >
                          <span className={`px-2 py-0.5 rounded-full text-xs font-medium ${roleColor(r)}`}>{r}</span>
                          {form.role === r && <Check className="w-3.5 h-3.5 text-indigo-600 ml-auto" />}
                        </button>
                      ))}
                    </div>
                  )}
                </div>
                <p className="text-xs text-gray-400 mt-1.5">
                  {form.role === 'owner'   && 'Full access to all settings, reports, and data.'}
                  {form.role === 'manager' && 'Can approve voids, apply discounts, and view reports.'}
                  {form.role === 'cashier' && 'Standard POS access: take orders and process payments.'}
                </p>
              </div>
            </div>

            <div className="flex gap-2 mt-6">
              <Button variant="outline" className="flex-1" onClick={closeModal} disabled={loading}>Cancel</Button>
              <Button className="flex-1 bg-indigo-600 hover:bg-indigo-700 text-white" onClick={handleSave} disabled={loading}>
                {loading ? (editingUser ? 'Savingâ€¦' : 'Creatingâ€¦') : (editingUser ? 'Save Changes' : 'Create User')}
              </Button>
            </div>
          </div>
        </div>
      )}

      {/* Reset Password Modal */}
      {resetUser && (
        <div className="fixed inset-0 z-50 flex items-center justify-center">
          <div className="absolute inset-0 bg-black/30 backdrop-blur-sm" onClick={() => setResetUser(null)} />
          <div className="relative bg-white rounded-2xl shadow-xl w-full max-w-sm mx-4 p-6">
            <div className="flex items-center justify-between mb-5">
              <div>
                <h4 className="text-base font-semibold text-gray-900">Reset Password</h4>
                <p className="text-xs text-gray-500 mt-0.5">Set a new password for <span className="font-medium text-gray-700">{resetUser.name}</span></p>
              </div>
              <button onClick={() => setResetUser(null)} className="p-1.5 rounded-lg hover:bg-gray-100 text-gray-400">
                <X className="w-4 h-4" />
              </button>
            </div>

            <div className="relative">
              <KeyRound className="absolute left-3 top-1/2 -translate-y-1/2 w-3.5 h-3.5 text-gray-400" />
              <Input
                type={showResetPassword ? 'text' : 'password'}
                value={resetPassword}
                onChange={e => setResetPassword(e.target.value)}
                placeholder="New password (min. 6 characters)"
                className="pl-9 pr-9"
                autoFocus
              />
              <button
                type="button"
                onClick={() => setShowResetPassword(p => !p)}
                className="absolute right-2.5 top-1/2 -translate-y-1/2 text-gray-400 hover:text-gray-600"
              >
                {showResetPassword ? <EyeOff className="w-3.5 h-3.5" /> : <Eye className="w-3.5 h-3.5" />}
              </button>
            </div>

            <div className="flex gap-2 mt-4">
              <Button variant="outline" className="flex-1" onClick={() => setResetUser(null)} disabled={resetLoading}>Cancel</Button>
              <Button className="flex-1 bg-indigo-600 hover:bg-indigo-700 text-white" onClick={handleResetPassword} disabled={resetLoading}>
                {resetLoading ? 'Resettingâ€¦' : 'Reset Password'}
              </Button>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}

export default function POSSettingsPage() {
  const supabase = createClient()
  const [shopId, setShopId] = useState('')
  const [tab, setTab] = useState<'terminals' | 'permissions' | 'printers' | 'users'>('terminals')

  useEffect(() => {
    supabase.from('shops').select('id').single().then(({ data }) => {
      if (data) setShopId(data.id)
    })
  }, [])

  const TABS = [
    { id: 'terminals',   label: 'POS Terminals',    icon: Monitor },
    { id: 'users',       label: 'User Management',  icon: UserPlus },
    { id: 'permissions', label: 'User Permissions', icon: Users },
    { id: 'printers',    label: 'Printer Control',  icon: Printer },
  ] as const

  return (
    <div className="p-8 max-w-3xl">
      <div className="mb-6">
        <h2 className="text-xl font-semibold text-gray-900">Users & POS Settings</h2>
        <p className="text-sm text-gray-500 mt-1">Manage POS terminals, user accounts, permissions, and printer assignments.</p>
      </div>

      {/* Tab bar */}
      <div className="flex gap-1 bg-gray-100 p-1 rounded-xl mb-6 w-fit">
        {TABS.map(({ id, label, icon: Icon }) => (
          <button
            key={id}
            onClick={() => setTab(id)}
            className={`flex items-center gap-1.5 px-4 py-2 rounded-lg text-sm font-medium transition-all ${
              tab === id ? 'bg-white text-gray-900 shadow-sm' : 'text-gray-500 hover:text-gray-700'
            }`}
          >
            <Icon className="w-3.5 h-3.5" />
            {label}
          </button>
        ))}
      </div>

      {tab === 'terminals'   && <TerminalsSection shopId={shopId} />}
      {tab === 'users'       && <UserManagementSection shopId={shopId} />}
      {tab === 'permissions' && <PermissionsSection shopId={shopId} />}
      {tab === 'printers'    && <PrinterControlSection shopId={shopId} />}
    </div>
  )
}
